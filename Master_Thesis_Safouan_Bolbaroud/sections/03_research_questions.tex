\chapter{Research questions}
\label{ch:research-questions}

\section{Main research question}
In mature open-source projects with documented security incidents, is there a traceable relationship between the rise of AI-generated code contributions and changes in technical debt, code review dynamics, and security outcomes?

\section{Sub-questions}
\begin{enumerate}
    \item \textbf{RQ1: Detection.} To what extent are AI-generated contributions identifiable in the selected projects, and what is their prevalence over time?
    \item \textbf{RQ2: Technical debt.} How have technical debt indicators and code quality metrics changed in the selected projects since the widespread availability of AI coding tools, and do AI-flagged contributions show different debt profiles than human-authored code?
    \item \textbf{RQ3: Security outcomes.} Can documented vulnerabilities be traced to contribution and review pathways involving AI-flagged code?
    \item \textbf{RQ4: Developer perspectives.} How do maintainers describe the effects of AI-assisted contributions on review burden, quality control, and security?
\end{enumerate}

\section{Analytical flow}
\textit{Explain how RQ1 establishes observability, RQ2 evaluates technical debt and quality patterns, RQ3 investigates security outcomes, and RQ4 interprets the mechanisms behind those findings.}
